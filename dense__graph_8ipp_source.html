37 DenseGraph::DenseGraph(const I &begin, const I &end) {
38  // Each VID in V is assumed to be (1) *unique* and (2) *between 0 and n-1*,
39  // given that the sequence is assumed to be *not* locally stored, it is
40  // necessary to *incrementally* resize the matrix when needed.
41  for (auto i = begin; i != end; i++) {
42  // Get the current matrix size.
43  VID n = order();
44  // Get the highest VID for the current EID, plus one to align the size.
45  VID v = (i->first > i->second ? i->first : i->second) + 1;
46  // Check if the VID exceeds the matrix size.
47  if (v > n) {
48  // Resize the matrix.
49  A.conservativeResize(v, v);
50  // Initialize the allocated memory to zero.
51  A(Eigen::all, Eigen::seq(n, Eigen::last)) *= 0;
52  A(Eigen::seq(n, Eigen::last), Eigen::all) *= 0;
53  }
54  // Set the edge.
55  A(i->first, i->second) = 1;
56  }
57 }

59 DenseGraph::DenseGraph(const AdjacencyList &other) {
60  // Given that the adjacency list *is* locally stored, it is possible to
61  // check whether each VID in V is (1) *unique* and (2) *between 0 and n-1* or not.
62 
63  // Check if the sequence is non-empty.
64  if (!other.empty()) {
65  // Get maximum VID in the sorted adjacency list and allign to size.
66  VID n = other.rbegin()->first + 1;
67  // Since the adjacency list is sorted, each VID in V is (1) *unique* by default,
68  // we still need to check if each VID is (2) *between 0 and n-1* or not.
69  if (n != other.size()) throw std::invalid_argument("AdjacencyList must contain VIDs *between 0 and size-1*.");
70  // Allocate a squared zero matrix.
71  A = AdjacencyMatrix::Zero(n, n);
72  // Fill the matrix.
73  for (const auto &[v, adj] : other) {
74  for (const auto &u : adj) {
75  // Check EID if is inconsistent with given VIDs.
76  if (other.find(u) == other.end()) throw std::out_of_range("AdjacencyList ill formed.");
77  // Set the edge.
78  A(v, u) = 1;
79  }
80  }
81  }
